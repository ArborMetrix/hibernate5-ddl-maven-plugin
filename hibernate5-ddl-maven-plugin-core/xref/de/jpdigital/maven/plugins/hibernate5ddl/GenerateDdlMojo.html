131     /**
132      * The Mojo's execute method.
133      *
134      * @throws MojoExecutionException if the Mojo can't be executed.
135      * @throws MojoFailureException   if something goes wrong while to Mojo is
136      *                                executed.
137      */
138     @Override
139     public void execute() throws MojoExecutionException, MojoFailureException {
140         final File outputDir = outputDirectory;
141 
142         getLog().info(String.format("Generating DDL SQL files in %s.",
143                                     outputDir.getAbsolutePath()));
144 
145         //Check if the output directory exists.
146         if (!outputDir.exists()) {
147             final boolean result = outputDir.mkdirs();
148             if (!result) {
149                 throw new MojoFailureException(
150                     "Failed to create output directory for SQL DDL files.");
151             }
152         }
153 
154         //Find the entity classes in the packages.
155         final Set<Class<?>> entityClasses = new HashSet<>();
156         for (final String packageName : packages) {
157             final Set<Class<?>> packageEntities = EntityFinder.forPackage(
158                 project, getLog(), packageName).findEntities();
159             entityClasses.addAll(packageEntities);
160         }
161         getLog().info(String.format("Found %d entities.",
162                                     entityClasses.size()));
163 
164         // Find the DDL generator implementation to use.
165         final ServiceLoader<DdlGenerator> serviceLoader = ServiceLoader
166             .load(DdlGenerator.class);
167         final DdlGenerator ddlGenerator;
168         if (serviceLoader.iterator().hasNext()) {
169             ddlGenerator = serviceLoader.iterator().next();
170         } else {
171             throw new MojoFailureException(String.format(
172                 "No implementation of '%s' is available.",
173                 DdlGenerator.class.getName()));
174         }
175 
176         for (final Dialect dialect : convertDialects()) {
177             ddlGenerator.generateDdl(dialect, entityClasses, this);
178         }
179 
180         if (customDialects != null) {
181             for (final String customDialect : customDialects) {
182                 ddlGenerator.generateDdl(customDialect, entityClasses, this);
183             }
184         }
185     }

285     /**
286      * Helper method which reads the dialects from the parameter and converts
287      * them into instances of the {@link Dialect} enumeration.
288      *
289      * @return A list of all dialects to use
290      *
291      * @throws MojoFailureException If an error occurs.
292      */
293     private Set<Dialect> convertDialects() throws MojoFailureException {
294 
295         final Set<Dialect> dialectsList = new HashSet<>();
296         if (dialects != null) {
297             for (final String dialect : dialects) {
298                 convertDialect(dialect, dialectsList);
299             }
300         }
301         return dialectsList;
302     }
303 
304     /**
305      * Helper method for converting the dialects from {@code String} to
306      * instances of the {@link Dialect} enumeration.
307      *
308      * @param dialect      The dialect to convert.
309      * @param dialectsList The lists of dialects where the converted dialect is
310      *                     stored.
311      *
312      * @throws MojoFailureException If the dialect string could not be
313      *                              converted, for example if it is misspelled.
314      *                              This will cause a {@code Build Failure}
315      */
316     private void convertDialect(final String dialect,
317                                 final Set<Dialect> dialectsList)
318         throws MojoFailureException {
319 
320         try {
321             dialectsList.add(Dialect
322                 .valueOf(dialect.toUpperCase(Locale.ENGLISH)));
323         } catch (IllegalArgumentException ex) {
324             final StringBuffer buffer = new StringBuffer();
325             for (final Dialect avilable : Dialect.values()) {
326                 buffer.append(avilable.toString()).append('\n');
327             }
328 
329             throw new MojoFailureException(
330                 String.format(
331                     "Can't convert the configured dialect '%s' to a dialect classname. "
332                     + "Available dialects are:%n"
333                         + "%s",
334                     dialect,
335                     buffer.toString()),
336                 ex);
337         }
338     }

340     protected void writeOutputFile(final String dialectClassName,
341                                    final Path tmpDir)
342         throws MojoFailureException {
343 
344         final OutputFileWriter writer = new OutputFileWriter(outputDirectory);
345         writer.setOmitDialectFromFileName(omitDialectFromFileName
346                                               && dialects.length == 1);
347         writer.setOutputFileNamePrefix(outputFileNamePrefix);
348         writer.setOutputFileNameSuffix(outputFileNameSuffix);
349 
350         writer.writeOutputFile(dialectClassName, tmpDir);
351 
352     }

354     public String getDialectNameFromClassName(final String dialectClassName) {
355 
356         final int pos = dialectClassName.lastIndexOf('.');
357 
358         if (dialectClassName.toLowerCase(Locale.ROOT).endsWith("dialect")) {
359             return dialectClassName
360                 .substring(pos + 1,
361                            dialectClassName.length() - "dialect".length())
362                 .toLowerCase(Locale.ROOT);
363         } else {
364             return dialectClassName.substring(pos + 1)
365                 .toLowerCase(Locale.ROOT);
366         }
367     }
